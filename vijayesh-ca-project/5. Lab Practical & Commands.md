## Lab Exercise 3: Brute Force Attack

### Objective
Crack passwords using brute force method.

### Step 1: Create Target Hash

```text
We want to crack: 1234 (MD5)
Hash: 81dc9bdb52d04dc20036dbd8313ed055
```

### Step 2: Brute Force with Mask

```bash
# Brute force 4 digit numbers
hashcat -m 0 -a 3 hashes.txt ?d?d?d?d

# Explanation:
# -m 0 = MD5
# -a 3 = Brute force
# ?d = digit (0-9)
# ?l = lowercase letter
# ?u = uppercase letter
# ?s = special character
# ?a = all characters
```

### Step 3: Brute Force Mixed Characters

```bash
# 4 character, lowercase + numbers
hashcat -m 0 -a 3 hashes.txt ?l?l?l?l
# Combinations: 36^4 = 1.6 million

# 4 character, all printable
hashcat -m 0 -a 3 hashes.txt ?a?a?a?a
# Combinations: 95^4 = 81 million
```

## Lab Exercise 5: Password Security Analysis

### Objective
Analyze a set of cracked passwords for security metrics.

### Step 1: Collect Passwords

```python
# passwords.py
passwords = [
    "123", "password", "123456", "admin", "letmein",
    "qwerty", "12345678", "1234567890", "abc123", "monkey",
    "master", "dragon", "111111", "baseball", "iloveyou",
    "trustno1", "sunshine", "princess", "welcome", "shadow"

```

### Step 2: Analyze Password Metrics

```python
# analyze.py
import re
from collections import Counter

passwords = ["123", "password", "123456", "admin", "letmein",
             "qwerty", "12345678", "1234567890", "abc123", "monkey"]

def analyze_passwords(passwords):
    results = {
        'total': len(passwords),
        'length_distribution': {},
        'character_types': {},
        'common_patterns': 0
    }
    
    for pwd in passwords:
        # Length analysis
        length = len(pwd)
        results['length_distribution'][length] = results['length_distribution'].get(length, 0) + 1
        
        # Character type analysis
        has_lower = bool(re.search(r'[a-z]', pwd))
        has_upper = bool(re.search(r'[A-Z]', pwd))
        has_digit = bool(re.search(r'[0-9]', pwd))
        has_special = bool(re.search(r'[!@#$%^&*]', pwd))
        
        # Common patterns
        if pwd in ['123', '1234', '12345', '123456', '1234567', '12345678']:
            results['common_patterns'] += 1
        if pwd in ['password', 'admin', 'letmein']:
            results['common_patterns'] += 1
    
    return results

results = analyze_passwords(passwords)
print("Password Security Analysis")
print("=" * 40)
print(f"Total Passwords: {results['total']}")
print(f"\nLength Distribution:")
for length, count in sorted(results['length_distribution'].items()):
    print(f"  {length} chars: {count}")
print(f"\nWeak Patterns: {results['common_patterns']}")
```

## Lab Exercise 6: Password Policy Testing

### Objective
Test password policy compliance.

### Step 1: Create Test Cases

```python
# policy_test.py

test_passwords = [
    ("123", False, "Too short"),
    ("password", False, "Common password"),
    ("P@ssw0rd1", True, "Complex but common"),
    ("CorrectHorseBatteryStaple", True, "Passphrase - GOOD"),
    ("Tr0ub4dor&3", True, "Complex but hard to remember"),
    ("aaaaaaaaaa", False, "Repeated characters"),
    ("1234567890qwerty", True, "Long - ACCEPTABLE"),
]

def test_password_policy(password, min_length=8, require_complex=True):
    issues = []
    
    if len(password) < min_length:
        issues.append(f"Too short (min {min_length})")
    
    if require_complex:
        has_upper = any(c.isupper() for c in password)
        has_lower = any(c.islower() for c in password)
        has_digit = any(c.isdigit() for c in password)
        has_special = any(c in "!@#$%^&*()" for c in password)
        
        if not (has_upper and has_lower and has_digit and has_special):
            issues.append("Missing complexity requirements")
    
    return len(issues) == 0, issues

print("Password Policy Test Results")
print("=" * 50)
for pwd, expected, desc in test_passwords:
    result, issues = test_password_policy(pwd)
    status = "✓ PASS" if result else "✗ FAIL"
    print(f"{status} | {pwd:25} | {desc}")
```

## Lab Exercise 7: Hash Generation for Testing

### Objective
Generate different hash types for testing.

### Step 1: Python Hash Generators

```python
import hashlib
import bcrypt
import secrets

# MD5 Hash
def hash_md5(password):
    return hashlib.md5(password.encode()).hexdigest()

# SHA-1 Hash
def hash_sha1(password):
    return hashlib.sha1(password.encode()).hexdigest()

# SHA-256 Hash
def hash_sha256(password):
    return hashlib.sha256(password.encode()).hexdigest()

# bcrypt Hash
def hash_bcrypt(password, work_factor=12):
    return bcrypt.hashpw(password.encode(), bcrypt.gensalt(rounds=work_factor)).decode()

# Test
test_password = "password123"
```

## Lab Exercise 8: Secure Password Storage

### Objective
Implement secure password storage.

### Step 1: Vulnerable Implementation (DON'T USE)

```python
# ❌ VULNERABLE - NEVER DO THIS
import hashlib

def register_user_unsafe(username, password):
    # No salt!
    password_hash = hashlib.md5(password.encode()).hexdigest()
    # Store directly in database
    save_to_database(username, password_hash)
```

### Step 2: Secure Implementation

```python
# ✅ SECURE - USE THIS
import bcrypt
import secrets

def register_user_secure(username, password):
    # Generate unique salt
    salt = secrets.token_hex(16)
    
    # Hash with salt using bcrypt
    password_hash = bcrypt.hashpw(password.encode(), bcrypt.gensalt(rounds=12))
    
    # Store in database
    save_to_database(username, password_hash.decode(), salt)

def verify_password(password, stored_hash):
    # Verify against stored hash
    return bcrypt.checkpw(password.encode(), stored_hash.encode())
```

## Lab Exercise 9: Password Strength Checker

### Objective
Build a password strength meter.

```python
import re

def check_password_strength(password):
    score = 0
    feedback = []
    
    # Length checks
    if len(password) < 8:
        feedback.append("❌ Too short (minimum 8 characters)")
    elif len(password) >= 12:
        score += 2
        feedback.append("✓ Good length")
    elif len(password) >= 8:
        score += 1
        feedback.append("✓ Acceptable length")
    
    # Character variety
    if re.search(r'[a-z]', password):
        score += 1
    if re.search(r'[A-Z]', password):
        score += 1
    if re.search(r'[0-9]', password):
        score += 1
    if re.search(r'[!@#$%^&*(),.?":{}|<>]', password):
        score += 1
    
    # Common password check
    common = ['password', '123456', 'qwerty', 'admin', 'letmein']
    if password.lower() in common:
        score = 0
        feedback.append("❌ Common password - easily cracked!")
    
    # Calculate strength
    if score >= 6:
        strength = "STRONG"
    elif score >= 4:
        strength = "MEDIUM"
    else:
        strength = "WEAK"
    
    return strength, score, feedback

# Test
test_passwords = ["123", "password", "P@ssw0rd", "CorrectHorseBatteryStaple"]

for pwd in test_passwords:
    strength, score, feedback = check_password_strength(pwd)
    print(f"\nPassword: {pwd}")
    print(f"Strength: {strength} (Score: {score}/7)")
    for f in feedback:
        print(f"  {f}")
```
